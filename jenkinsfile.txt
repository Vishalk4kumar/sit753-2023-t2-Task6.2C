pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                echo 'Mavens'
            }
        }
        stage('Unit and Integration Tests') {
            steps {
                echo 'JUnit (for unit tests)'
                echo 'Postman (for integration tests)'
            }
            post {
                success {
                    script {
                        emailext (
                            subject: "Unit and Integration Tests Passed",
                            body: "Unit and Integration Tests have passed successfully.",
                            attachLog: true,
                            to: 'lohana.vishal@gmail.com'
                        )
                    }
                }
                failure {
                    script {
                        emailext (
                            subject: "Unit and Integration Tests Failed",
                            body: "Unit and Integration Tests have failed.",
                            attachLog: true,
                            to: 'lohana.vishal@gmail.com'
                        )
                    }
                }
            }
        }
        stage('Code Analysis') {
            steps {
                echo 'SonarQube'
            }
        }
        stage('Security Scan') {
            steps {
                echo 'OWASP ZAP'
            }
            post {
                success {
                    script {
                        emailext (
                            subject: "Security Scan Passed",
                            body: "Security Scan has passed successfully.",
                            attachLog: true,
                            to: 'lohana.vishal@gmail.com'
                        )
                    }
                }
                failure {
                    script {
                        emailext (
                            subject: "Security Scan Failed",
                            body: "Security Scan has failed.",
                            attachLog: true,
                            to: 'lohana.vishal@gmail.com'
                        )
                    }
                }
            }
        }
        stage('Deploy to Staging') {
            steps {
                echo 'Ansible'
            }
        }
        stage('Integration Tests on Staging') {
            steps {
                echo 'Selenium'
            }
        }
        stage('Deploy to Production') {
            steps {
                echo 'Jenkins'
            }
        }
    }
}
